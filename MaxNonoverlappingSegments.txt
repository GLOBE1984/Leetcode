class Solution {
    public int solution(int[] A, int[] B) {
        // write your code in Java SE 8
        int length = A.length;
        
        int index_last = 0;
        int count = 0;
        for(int i = 0; i < length; ++i) {
            if(i == 0) {
                index_last = B[0];
                count++;
            }
            else {
                if(A[i] > index_last) {
                    count++;
                    index_last = B[i];
                }
            }
        }
        return count;
    }
